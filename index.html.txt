<script>
  let burgers = JSON.parse(localStorage.getItem('burgers') || '[]');
  let rating = 0;

  function init() {
    document.getElementById('f-date').value = new Date().toISOString().split('T')[0];
    setupStars();
    renderVault();
    renderStats();
  }

  function save() {
    localStorage.setItem('burgers', JSON.stringify(burgers));
  }

  function setupStars() {
    document.querySelectorAll('.star-btn').forEach(s => {
      s.addEventListener('click', () => {
        rating = parseInt(s.dataset.v);
        updateStars(rating);
      });
      s.addEventListener('mouseover', () => updateStars(parseInt(s.dataset.v)));
      s.addEventListener('mouseout', () => updateStars(rating));
    });
  }

  function updateStars(n) {
    document.querySelectorAll('.star-btn').forEach(b => {
      b.classList.toggle('lit', parseInt(b.dataset.v) <= n);
    });
  }

  function starsHTML(n) {
    return '⭐'.repeat(n) + '<span style="opacity:0.25">⭐</span>'.repeat(5 - n);
  }

  function switchTab(t) {
    const tabs = ['vault', 'log', 'stats'];
    document.querySelectorAll('.tab').forEach(el => el.classList.remove('active'));
    document.querySelectorAll('.view').forEach(el => el.classList.remove('active'));
    document.querySelectorAll('.tab')[tabs.indexOf(t)].classList.add('active');
    document.getElementById('view-' + t).classList.add('active');
    if (t === 'stats') renderStats();
    if (t === 'vault') renderVault();
  }

  function addBurger() {
    const name = document.getElementById('f-name').value.trim();
    if (!name) { document.getElementById('f-name').focus(); return; }
    burgers.unshift({
      id: Date.now(),
      name,
      loc: document.getElementById('f-loc').value.trim() || 'Unknown spot',
      date: document.getElementById('f-date').value,
      style: document.getElementById('f-style').value,
      rating: rating || 3,
      notes: document.getElementById('f-notes').value.trim()
    });
    save();
    document.getElementById('f-name').value = '';
    document.getElementById('f-loc').value = '';
    document.getElementById('f-notes').value = '';
    document.getElementById('f-style').value = 'smash';
    document.getElementById('f-date').value = new Date().toISOString().split('T')[0];
    rating = 0;
    updateStars(0);
    renderVault();
    switchTab('vault');
  }

  function deleteBurger(id) {
    burgers = burgers.filter(b => b.id !== id);
    save();
    renderVault();
    renderStats();
  }

  function esc(s) {
    return String(s)
      .replace(/&/g, '&amp;').replace(/</g, '&lt;')
      .replace(/>/g, '&gt;').replace(/"/g, '&quot;');
  }

  function renderVault() {
    const el = document.getElementById('vault-content');
    if (!burgers.length) {
      el.innerHTML = `<div class="empty"><div class="empty-icon">🍔</div><div>No burgers logged yet.</div><div style="margin-top:6px;font-size:12px;">Hit the Log tab to record your first one.</div></div>`;
      return;
    }
    const sorted = [...burgers].sort((a, b) => new Date(b.date) - new Date(a.date));
    el.innerHTML = `<div class="section-head" style="margin-bottom:12px;">${burgers.length} burger${burgers.length !== 1 ? 's' : ''} in the vault</div>
    <div class="burger-list">${sorted.map(b => `
      <div class="burger-card">
        <div class="burger-icon">🍔</div>
        <div class="burger-info">
          <div class="burger-name">${esc(b.name)}</div>
          <div class="burger-meta">${esc(b.loc)} · ${b.date}<span class="tag tag-style">${b.style}</span></div>
          <div style="margin-top:4px;">${starsHTML(b.rating)}</div>
          ${b.notes ? `<div class="burger-note">"${esc(b.notes)}"</div>` : ''}
        </div>
        <button class="delete-btn" onclick="deleteBurger(${b.id})" title="Delete">×</button>
      </div>`).join('')}</div>`;
  }

  function renderStats() {
    const el = document.getElementById('stats-content');
    if (!burgers.length) {
      el.innerHTML = `<div class="empty"><div class="empty-icon">📊</div><div>Log some burgers to see stats.</div></div>`;
      return;
    }
    const total = burgers.length;
    const avg = burgers.reduce((s, b) => s + b.rating, 0) / total;
    const best = burgers.reduce((a, b) => b.rating > a.rating ? b : a, burgers[0]);
    const dist = [1, 2, 3, 4, 5].map(n => burgers.filter(b => b.rating === n).length);
    const maxD = Math.max(...dist, 1);
    const styles = {};
    burgers.forEach(b => styles[b.style] = (styles[b.style] || 0) + 1);
    const topStyle = Object.entries(styles).sort((a, b) => b[1] - a[1])[0];

    el.innerHTML = `
    <div class="stats-grid">
      <div class="stat"><div class="stat-val">${total}</div><div class="stat-label">burgers logged</div></div>
      <div class="stat"><div class="stat-val" style="color:var(--obs-yellow);">${avg.toFixed(1)}</div><div class="stat-label">avg rating</div></div>
      <div class="stat"><div class="stat-val" style="color:var(--obs-teal);">${topStyle[0]}</div><div class="stat-label">fave style</div></div>
    </div>
    <div class="best-badge"><strong>🏆 Top-rated: ${esc(best.name)}</strong>${esc(best.loc)} · ${'⭐'.repeat(best.rating)}</div>
    <div class="section-head">Rating distribution</div>
    <div class="chart-bar-wrap">${[5, 4, 3, 2, 1].map(n => `
      <div class="chart-row">
        <div class="chart-label">${n}</div>
        <div class="chart-track"><div class="chart-fill" style="width:${Math.round(dist[n-1] / maxD * 100)}%"></div></div>
        <div class="chart-count">${dist[n - 1]}</div>
      </div>`).join('')}</div>`;
  }

  init();
</script>
